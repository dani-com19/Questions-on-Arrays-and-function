#include <iostream>
using namespace std;

// Function to calculate the square of a number
int square(int num) {
    return num * num;
}

int main() {
    int number;

    // Ask the user to enter a number
    cout << "Enter a number: ";
    cin >> number;

    // Call the square function and display the result
    int result = square(number);
    cout << "The square of " << number << " is: " << result << endl;

    return 0;
}
