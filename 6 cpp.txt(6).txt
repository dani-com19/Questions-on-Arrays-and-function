#include <iostream>
using namespace std;

// Function to calculate base^exponent using a loop
int power(int base, int exponent) {
    int result = 1;

    // Multiply base, exponent times
    for (int i = 1; i <= exponent; i++) {
        result *= base;
    }

    return result;
}

int main() {
    int base, exponent;

    // Get user input
    cout << "Enter the base: ";
    cin >> base;

    cout << "Enter the exponent: ";
    cin >> exponent;

    // Call power function and display result
    int result = power(base, exponent);
    cout << base << "^" << exponent << " = " << result << endl;

    return 0;
}
